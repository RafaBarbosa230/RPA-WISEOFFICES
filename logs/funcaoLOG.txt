def configurar_log():
    log_format = "%(asctime)s - %(levelname)s - %(message)s"
    log_file = f"logs/automacao_{datetime.now().strftime('%Y-%m-%d')}.log"  
    logging.basicConfig(
        level=logging.DEBUG,        
        format=log_format,
        handlers=[
            logging.FileHandler(log_file),  
            logging.StreamHandler()         
        ]
    )

    # Informa que o log foi configurado
    logging.info('Log configurado com sucesso!')
configurar_log()
